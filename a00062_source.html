160 void executionSystemClass::ModuleExeArea(int EXE_AREA_INDEX)
161 {
162  if(exceptionListHead!=nullptr)
163  {
164  linkedEntryPointClass* currentExeNode;
165  switch(EXE_AREA_INDEX)
166  {
167  case EXP_SETUP:
168  currentExeNode = setupListHead;
169  break;
170  case EXP_LOOP:
171  currentExeNode = loopListHead;
172  break;
173  case EXP_SYSTICK:
174  currentExeNode = sysTickListHead;
175  break;
176  default:
177  return;
178  }
179 
180  if(currentExeNode!=nullptr)
181  {
182  int retVal;
183  do
184  {
185 
186 #ifndef __NOEXCEPTIONS
187  try
188  {
189 #endif
190  if(currentExeNode->getComputeModule()->getModuleDataPtr()->exceptionFlags == 0u)
191  {
192  switch(EXE_AREA_INDEX)
193  {
194  case EXP_SETUP:
195  retVal = currentExeNode->getComputeModule()->mod_setup();
196  break;
197  case EXP_LOOP:

199  break;
200  case EXP_SYSTICK:

213  }
214  catch(...)
215  {
216 #endif

218 #ifndef __NOEXCEPTIONS
219  }
220 #endif
221  if(currentExeNode->getNextEPClassPtr() != nullptr)
222  currentExeNode = currentExeNode->getNextEPClassPtr();
