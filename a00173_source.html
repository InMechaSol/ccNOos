70 void platformStart()
71 {
72  //<platformStart>
73  /* Configure the SysTick timer to generate interrupt every 1 ms
74  * and start its operation. Call the function before assigning the
75  * callbacks as it calls CySysTickInit() during first run that
76  * re-initializes the callback addresses to the NULL pointers.
77  */
78  CySysTickStart();
79 
80  /* Find unused callback slot and assign the callback. */
81  for (int i = 0u; i < (int)CY_SYS_SYST_NUM_OF_CALLBACKS; ++i)
82  {
83  if (CySysTickGetCallback(i) == NULL)
84  {
85  /* Set callback */
86  CySysTickSetCallback(i, SysTickISRCallback);
87  break;
88  }
89  }
90  UART_Start();
91  //</platformStart>
92 }
